function doPost(e){
  var ss=SpreadsheetApp.getActiveSpreadsheet();
  var data=JSON.parse(e.postData.contents);
  if(data.action==='save'){
    saveData(ss,data.payload);
    return ok();
  }
  return ok();
}
function doGet(e){
  var ss=SpreadsheetApp.getActiveSpreadsheet();
  var sh=ss.getSheetByName('AppData');
  if(!sh)return ContentService.createTextOutput(JSON.stringify({data:null})).setMimeType(ContentService.MimeType.JSON);
  var val=sh.getRange('A1').getValue();
  return ContentService.createTextOutput(val||'{}').setMimeType(ContentService.MimeType.JSON);
}
function saveData(ss,payload){
  var sh=ss.getSheetByName('AppData')||ss.insertSheet('AppData');
  sh.getRange('A1').setValue(JSON.stringify(payload));
  writeReadable(ss,payload);
}
function writeReadable(ss,d){
  var sh=ss.getSheetByName('Readable')||ss.insertSheet('Readable');
  sh.clearContents();
  var rows=[['MATAJI MANDIR — READABLE DATA','',''],['','',''],['SADASY LIST','',''],['#','Naam','Default Rs.']];
  (d.masters||[]).forEach(function(m,i){rows.push([i+1,m.name,m.defAmt]);});
  rows.push(['','',''],['MONTHLY DATA','','']);
  Object.keys(d.monthly||{}).sort().forEach(function(mk){
    rows.push(['--- '+mk.replace('_',' ')+' ---','','']);
    rows.push(['#','Naam','Status','Amount']);
    (d.monthly[mk]||[]).forEach(function(c,i){rows.push([i+1,c.name,c.status,c.amount]);});
    rows.push(['','','','']);
  });
  rows.push(['KHARCHA','','','']);
  rows.push(['Vivran','Tarikh','Amount','Month']);
  (d.expenses||[]).forEach(function(e){rows.push([e.name,e.date,e.amount,e.month]);});
  sh.getRange(1,1,rows.length,4).setValues(rows);
}
function ok(){return ContentService.createTextOutput(JSON.stringify({status:'ok'})).setMimeType(ContentService.MimeType.JSON);}
